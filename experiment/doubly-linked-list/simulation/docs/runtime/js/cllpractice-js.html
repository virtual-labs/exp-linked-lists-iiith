1 Introduction

This is interactive module to learn Circular Linked List basic concepts. User can see simple representation of Circular Linked List User can click on one of the three operations: Insert, Search, Remove Insert operation has further three more options: Insert at head Insert at tail Insert afer a node

2 Circular Linked List Practice Artefact

The artefact is implemented in Javascript using Canvas module. No other external libraries are used.

3.8 Draw Box

  function drawBox(x, y, ind, color) {
  
    val = numbers[ind];
  
    if (color==ncolor)
        ctx.drawImage(box, x, y, nWidth, nHeight);
    else if (color==scolor)
        ctx.drawImage(sbox, x, y, nWidth, nHeight);
    else
        ctx.drawImage(tbox, x, y, nWidth, nHeight);
  
    ctx.globalAlpha = 0.5;
    ctx.drawImage(boxline, x+linegap*.9, y+(nHeight-lineheight)/2, linewidth, lineheight);
    ctx.globalAlpha = 1;
  
    if (ind==numbers.length-1)  
    {
        ctx.beginPath();
        ctx.lineWidth = 1;
        ctx.strokeStyle = "#979091";
        ctx.moveTo(x+(linegap+15)*.9+1, y+(nHeight-dotheight)/2);
        ctx.lineTo(x+(linegap+15)*.9+50, (rectHeight/2)+y+10);
        ctx.lineTo(x+(linegap+15)*.9+50, (rectHeight/2)+y+65);
        ctx.lineTo(65, (rectHeight/2)+y+65);
        ctx.lineTo(65, (rectHeight/2)+y+10);
        ctx.stroke();
  
        ctx.closePath();
        drawArrow(65, (rectHeight/2)+y+10, 120, (rectHeight/2)+y+10);
    }
    else
    {
        ctx.drawImage(arrowtriangle, x+(linegap+15+43)*.9, y+(nHeight-dotheight)/2-5, 20, 14);
        ctx.drawImage(arrowline, x+linegap+15, y+(nHeight-dotheight)/2, 55*0.6, 5);
    }
  
    ctx.drawImage(dot, x+(linegap+15)*.9, y+(nHeight-dotheight)/2-5, dotwidth, dotheight);
  
    ctx.beginPath();
    if (color==ncolor)
        ctx.fillStyle =  "#2180bc";
    else if (color==scolor)
        ctx.fillStyle =  "#fff";
    else
        ctx.fillStyle =  "#fff";

  }

3.9 Draw Arrow

The function to draw an arrow using canvas 

function drawArrow(startX, startY, endX, endY) {

        ctx.beginPath();
    var headlen = 20;   // length of head in pixels
    var angle = Math.atan2(endY-startY,endX-startX);

        ctx.strokeStyle = "#d2d2d2";
    ctx.moveTo(startX, startY);
    ctx.lineTo(endX, endY);
    ctx.stroke();
    ctx.closePath();

    ctx.beginPath();
    ctx.moveTo(endX-headlen*Math.cos(angle-Math.PI/6),endY-headlen*Math.sin(angle-Math.PI/6));
    ctx.lineTo(endX, endY);
    ctx.lineTo(endX-headlen*Math.cos(angle+Math.PI/6),endY-headlen*Math.sin(angle+Math.PI/6));
    ctx.fillStyle = "#979091";
    ctx.fill();
    ctx.stroke();
        ctx.closePath();
}

function draw_head(){

3.11 Renderer Function

Main function which refreshes linked list on screen 

function renderer(){

        ctx.clearRect(0, 0, 1360, 320);
        draw_head();
        drawArrow(headX, headY, headX, topgap);         

        for(var i=0; i<numbers.length-1; i++) {
                drawBox(leftgap + (i*boxDist), topgap, i, ncolor);
        }
        
        if (numbers.length>0)
                drawBox(leftgap + (numbers.length-1)*(boxDist), topgap, numbers.length-1, ncolor);                                              
}

3.12 Array Maker

Function which makes changes to original numbers array 

function array_maker(){

        if (decider==1)
                numbers.push(value);
        if (decider==2)
                numbers.insert(index,value);    
        if (decider==3)
                numbers.splice(index-1, 1);
        if (decider==4)
                numbers.unshift(value);

}

3.13 Linked List Node Shifter

The function which shows a node being traversed,main animation function 

function nodeshift(){


        if (keyc==boxDist)
        {
                busy = 0;
                clearInterval(shift_stopper);
                array_maker();
                renderer();
                return;
        }

        if (keyc==-boxDist)
        {
                busy = 0;
                clearInterval(shift_stopper);
                array_maker();
                renderer();
                return;
        }

        ctx.clearRect(0, 0, 1360, 320);
        draw_head();
        if (decider==4) 
                drawArrow(headX, headY, headX+keyc, topgap);    
        else
                drawArrow(headX, headY, headX, topgap); 

        for(var i=0; i<index-1; i++) {
                drawBox(leftgap + (i*boxDist), topgap, i, ncolor);
        }

        if (decider==2)
        {
                i = index - 1;
                drawBox(leftgap + (i*boxDist), topgap, i, ncolor);
        }                       

        for (var i=index; i<numbers.length; i++){
                drawBox(leftgap + (i*boxDist)+keyc, topgap, i, ncolor);
        }
        if (decider==3)
                keyc = keyc - 1;
        else
                keyc = keyc + 1;

}

3.14 Colorer

The function which shows color animation 

function colorer(last){

  if (numa==last)
  {
        if (decider==1)
        {       
          array_maker();
          busy = 0;
          clearInterval(color_stopper);
          renderer();
        }

        if (decider==2)
        {
          shift_stopper = setInterval(nodeshift, 1);
          setTimeout(function(){ clearInterval(color_stopper);  busy = 0;}, boxDist);
        }
        return;
  }

  ctx.clearRect(0, 0, 1360, 320);
  renderer();
  for(var i=0; i<numbers.length; i++) {
        if (i!=numa)
          drawBox(leftgap + (i*boxDist), topgap, i, ncolor);
        else
          drawBox(leftgap + (i*boxDist), topgap, i, dcolor);
  }
  numa = numa + 1;
}

3.15 Search

The function used for searching a node 

function searcher(val){

  searchover = 0;

  if (numb==numbers.length)
  {
        searchover = 1;
        clearInterval(color_stopperb);
        busy = 0;
        renderer();
        document.getElementById('ins').innerHTML= 'Value not found!';
        return;
  }

  ctx.clearRect(0, 0, 1360, 320);
  draw_head();
  drawArrow(headX, headY, headX, topgap);       

  for(var i=0; i<numbers.length-1; i++) {
        if (i!=numb)
          drawBox(leftgap + (i*boxDist), topgap, i, ncolor);
        else
        {
          if (numbers[i]==val)
          {
                searchover = 1;
                drawBox(leftgap + (i*boxDist), topgap, i, scolor);
                document.getElementById('ins').innerHTML= 'Value found!';
                clearInterval(color_stopperb);
                if (decider==5)
                  busy = 0;

          }
          else
                drawBox(leftgap + (i*boxDist), topgap, i, dcolor);
        }
  }
  
  if (numb!=numbers.length-1)
        drawBox(leftgap + (numbers.length-1)*(boxDist), topgap, numbers.length-1, ncolor);      
  else
  {
        if (numbers[numb]==val)
        {
          searchover = 1;
          drawBox(leftgap + (numbers.length-1)*(boxDist), topgap, i, scolor);
          clearInterval(color_stopperb);
          document.getElementById('ins').innerHTML= 'Value found!';
          if (decider==5)
                busy = 0;

        }
        else
          drawBox(leftgap + (numbers.length-1)*(boxDist), topgap, i, dcolor);
  }

  if ((searchover==1)&&(decider==3))
  {
        if (index!=0)
        {
          setTimeout(function(){ shift_stopper = setInterval(nodeshift, 1);
                                                   }, 1000);
        }
  }
  numb = numb + 1;
}

3.16.1 Insert At Head

function insertAtHead() {
        if (busy==1){
                clear();
                return;
        }
        else
                busy=1;
        num = 0;
        value = document.getElementById('HeadtoBeInserted').value;
        index = 0;

}
