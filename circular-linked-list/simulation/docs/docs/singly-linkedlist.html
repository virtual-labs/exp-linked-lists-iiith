1 Introduction

Here is the javascript implementation of singly linked list with following operations:

1.1 Insertion

  • Insertion at head
  • Insertion at tail
  • Insertion after a given previous node

1.2 Search

  • Given a key search for it's occurence throughout the linked list.

1.3 Delete

  • Delete the first occurence of a given key in linked list

1.4 Display Linked List

  • The linked list will be displayed from head to tail

3 Node Constructor

Function node() acts as constructor of a single node. Any instance of node can be created by using operator new with node() and supplying data to be stored in node as parameter. 

            function node(val) {
            this.val = val;
            this.next = null;
        };

                this.head = newnode;
            
            while(temp.next)

            temp.next = newnode;

7 Insert after a given node

            this.insert_after = function(prevnode,val) {
            var newnode = new node(val);
                newnode.next = prevnode.next;
                prevnode.next = newnode;
           }

8 Search

            this.search = function(val) {
            var temp = this.head; 
            while(temp) {
                if (temp.val==val)
                {
                   console.log('The key is there in linked list');      
                   return temp;
                }
                temp = temp.next;
            }
            console.log('The key is not there in linked list');         
            return temp;
          }

9 Delete

            this.remove = function(val) {
            var temp = this.head; 
            var flag = 0;
            var prev = null;
            if (temp != null && temp.val==val)
            {
                this.head = temp.next;
                flag = 1;
                console.log('The key is deleted');
                return flag;
            }

            while (temp != null && temp.val != val)
            {
                prev = temp;
                temp = temp.next;
            }

            if (temp == null)
            {
                console.log('The key is not there in linked list');
                return flag;
            }
            prev.next = temp.next;
            flag = 1;
            console.log('The key is deleted');
            return flag;
          }
